name: CI Tests

on:
  push:
    branches: [main, develop]
    paths:
      - 'scripts/**'
      - '.github/workflows/test.yml'
  pull_request:
    branches: [main]
    paths:
      - 'scripts/**'
      - '.github/workflows/test.yml'
  workflow_dispatch:

permissions:
  contents: read

jobs:
  shellcheck:
    name: ShellCheck Lint
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Install ShellCheck
        run: |
          sudo apt-get update -qq
          sudo apt-get install -y -qq shellcheck

      - name: Run ShellCheck on all scripts
        run: |
          echo "=== ShellCheck Analysis ==="
          find scripts/ -name '*.sh' -type f -print0 | \
            xargs -0 shellcheck --severity=warning --format=gcc
          echo "ShellCheck passed."

  syntax-check:
    name: Bash Syntax Validation
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Validate bash syntax
        run: |
          echo "=== Bash Syntax Check ==="
          errors=0
          for script in scripts/*.sh; do
            echo "Checking: ${script}"
            if ! bash -n "$script" 2>&1; then
              echo "FAIL: ${script}"
              errors=$((errors + 1))
            else
              echo "  OK"
            fi
          done
          if [[ "$errors" -gt 0 ]]; then
            echo "FAILED: ${errors} script(s) have syntax errors."
            exit 1
          fi
          echo "All scripts passed syntax validation."

  permissions-check:
    name: File Permissions Audit
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Verify script permissions
        run: |
          echo "=== Permissions Audit ==="
          errors=0
          for script in scripts/*.sh; do
            if [[ ! -x "$script" ]]; then
              echo "WARNING: ${script} is not executable. Run: chmod +x ${script}"
              errors=$((errors + 1))
            else
              echo "  OK: ${script} (+x)"
            fi
          done
          if [[ "$errors" -gt 0 ]]; then
            echo ""
            echo "NOTE: ${errors} script(s) missing execute permission."
            echo "Fix with: chmod +x scripts/*.sh"
            exit 1
          fi
          echo "All scripts have correct permissions."

  distro-matrix:
    name: "Test on ${{ matrix.distro }}"
    runs-on: ubuntu-latest
    needs: [shellcheck, syntax-check]
    strategy:
      fail-fast: false
      matrix:
        include:
          - distro: ubuntu:22.04
            label: Ubuntu 22.04
          - distro: ubuntu:24.04
            label: Ubuntu 24.04
          - distro: debian:12
            label: Debian 12
          - distro: rockylinux:9
            label: Rocky Linux 9
          - distro: almalinux:9
            label: AlmaLinux 9
          - distro: fedora:40
            label: Fedora 40
          - distro: amazonlinux:2023
            label: Amazon Linux 2023

    container:
      image: ${{ matrix.distro }}

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Install bash (if needed)
        run: |
          if command -v apt-get &>/dev/null; then
            apt-get update -qq && apt-get install -y -qq bash coreutils
          elif command -v dnf &>/dev/null; then
            dnf install -y -q bash coreutils
          elif command -v yum &>/dev/null; then
            yum install -y -q bash coreutils
          fi

      - name: Validate script syntax
        run: |
          for script in scripts/*.sh; do
            echo "[${{ matrix.label }}] Checking: ${script}"
            bash -n "$script"
          done
          echo "[${{ matrix.label }}] All scripts passed."

      - name: Test help output
        run: |
          echo "--- Cron script help ---"
          bash scripts/deploy-mde-cron.sh --help || true
          echo ""
          echo "--- Systemd script help ---"
          bash scripts/deploy-mde-systemd.sh --help || true

      - name: Test dry-run mode
        run: |
          echo "--- Cron dry-run ---"
          bash scripts/deploy-mde-cron.sh --dry-run 2>&1 || true
          echo ""
          echo "--- Systemd dry-run ---"
          bash scripts/deploy-mde-systemd.sh --dry-run 2>&1 || true
